#include <iostream>
using namespace std;

class Stack {
private:
    int stack_size;
    int array_capacity;
    int* array;

public:
    Stack(int n = 10) {
        if (n <= 0) {
            cout << "Enter positive number" << endl;
            return;
        }
        array_capacity = n;
        stack_size = 0;
        array = new int[n];
    }

    ~Stack() {
        delete[] array;
    }

    bool empty() const {
        return (stack_size == 0);
    }

    bool full() const {
        return (stack_size == array_capacity);
    }

    int top() const {
        if (empty()) {
            throw "Underflow";
        }
        return array[stack_size - 1];
    }

    int pop() {
        if (empty()) {
            throw "Underflow";
        }
        --stack_size;
        return array[stack_size];
    }

    void push(int obj) {
        if (full()) {
            throw "Overflow";
        }
        array[stack_size] = obj;
        ++stack_size;
    }

    void display() {
        cout << "Stack (bottom â†’ top): ";
        for (int i = 0; i < stack_size; i++) {
            cout << array[i] << " ";
        }
        cout << endl;
    }
};

int main() {
    Stack s(-5);   // test negative

    Stack s2(5);
    s2.push(10);
    s2.push(20);
    s2.push(30);
    s2.display();

    return 0;
}
